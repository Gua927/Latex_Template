\subsection{Mathematics}
\begin{frame}{Mathematics}{Equations, Theorems and Proofs}
\begin{block}{Key Equation}
    The standard diffusion forward process:
    \[
        q(\mathbf{z}_t \mid \mathbf{z}_0) = \mathcal{N}(\alpha_t \mathbf{z}_0, \sigma_t^2 \mathbf{I}), \quad \mathbf{z}_0 \in \R^d.
    \]
\end{block}

\vspace{-0.15cm}
\begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{theorem}[Fenchel--Young Inequality]
        For any $x, y \in \R^n$ and convex $f$:
        \[
            f(x) + f^*(y) \geq \langle x, y \rangle.
        \]
    \end{theorem}

    \column{0.48\textwidth}
    \begin{definition}[Lipschitz Smoothness]
        A function $f \colon \R^d \to \R$ is $L$-smooth if
        \[
            \|\nabla f(x) - \nabla f(y)\| \leq L\|x - y\|.
        \]
    \end{definition}
\end{columns}

\vspace{0.1cm}
\begin{proof}
    By Jensen's inequality and the convexity of $f$, we have $f\bigl(\E[X]\bigr) \leq \E[f(X)]$ for any random variable $X$. \qedhere
\end{proof}
